validerRelance: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'validerRelance', relanceId: this.selectedRelance?.id }); this.loading = true; try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { endpoint: `/api/relances/${this.selectedRelance.id}/valider` }); const response = await fetch(`/api/relances/${this.selectedRelance.id}/valider`, { method: 'POST', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error(`Erreur HTTP ${response.status}`); // Retirer de la liste this.relances = this.relances.filter(r => r.id !== this.selectedRelance.id); this.selectedRelance = null; this.editForm = { destinataire: '', date: '', objet: '', corps: '' }; log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
