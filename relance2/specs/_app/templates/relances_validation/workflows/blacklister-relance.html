blacklisterRelance: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'blacklisterRelance', relanceId: this.selectedRelance?.id }); if (!confirm('Blacklister ce payeur ? Aucune relance ne lui sera plus envoyée.')) { log.info('WORKFLOW_CANCELLED', { workflowId, reason: 'user_cancelled' }); return; } this.loading = true; try { const token = localStorage.getItem('token'); const payeur = this.selectedRelance.payeur; log.debug('API_CALL_START', { endpoint: '/api/payeurs/blacklist' }); const response = await fetch('/api/payeurs/blacklist', { method: 'POST', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ payeur }) }); if (!response.ok) throw new Error(`Erreur HTTP ${response.status}`); // Retirer toutes les relances de ce payeur this.relances = this.relances.filter(r => r.payeur !== payeur); this.selectedRelance = null; log.info('WORKFLOW_SUCCESS', { workflowId, payeur }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
