saveChanges: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'saveChanges', relanceId: this.selectedRelance?.id }); this.loading = true; try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { endpoint: `/api/relances/${this.selectedRelance.id}` }); const response = await fetch(`/api/relances/${this.selectedRelance.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify(this.editForm) }); if (!response.ok) throw new Error(`Erreur HTTP ${response.status}`); // Mise à jour locale const idx = this.relances.findIndex(r => r.id === this.selectedRelance.id); if (idx !== -1) { this.relances[idx] = { ...this.relances[idx], ...this.editForm }; } log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
