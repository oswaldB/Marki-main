initialLoad: async function() { const workflowId = crypto.randomUUID(); const startTime = Date.now(); log.info('WORKFLOW_START', { workflowId, workflow: 'initialLoad', page: 'relances_detail', relanceId: this.relanceId }); this.loading = true; this.error = null; try { // Étape 1: Vérification auth const token = localStorage.getItem('token'); if (!token) { log.warn('AUTH_MISSING'); window.location.href = '/login'; return; } log.debug('AUTH_VERIFIED'); // Étape 2: Appel API log.debug('API_CALL_START', { endpoint: `/api/relances/${this.relanceId}` }); const response = await fetch(`/api/relances/${this.relanceId}`, { headers: { 'Authorization': `Bearer ${token}` } }); log.debug('API_RESPONSE_RECEIVED', { status: response.status }); if (!response.ok) { if (response.status === 404) { throw new Error('Relance non trouvée'); } throw new Error(`Erreur HTTP ${response.status}`); } const data = await response.json(); this.relance = data; log.debug('DATA_LOADED', { relanceNom: this.relance.nom }); log.info('WORKFLOW_SUCCESS', { workflowId, duration: Date.now() - startTime }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message, stack: error.stack }); } finally { this.loading = false; } }
