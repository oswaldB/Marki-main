duplicateSequence: async function(sequence) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'duplicateSequence', sequenceId: sequence?.id }); this.loading = true; try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { endpoint: `/api/sequences/${sequence.id}/duplicate` }); const response = await fetch(`/api/sequences/${sequence.id}/duplicate`, { method: 'POST', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error(`Erreur HTTP ${response.status}`); const duplicated = await response.json(); this.sequences.push(duplicated); log.info('WORKFLOW_SUCCESS', { workflowId, newSequenceId: duplicated.id }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
