downloadFacture: async function(id) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'downloadFacture', factureId: id }); try { window.open(`/api/portail/factures/${id}/pdf?token=${this.token}`, '_blank'); log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
