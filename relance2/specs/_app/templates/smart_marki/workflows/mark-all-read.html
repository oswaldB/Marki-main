markAllRead: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'markAllRead' }); this.loading = true; try { const token = localStorage.getItem('token'); const unreadIds = this.insights.filter(i => !i.lu).map(i => i.id); if (unreadIds.length === 0) { log.info('WORKFLOW_SKIP', { workflowId, reason: 'no_unread_insights' }); return; } log.debug('API_CALL_START', { endpoint: '/api/insights/mark-all-read', count: unreadIds.length }); const response = await fetch('/api/insights/mark-all-read', { method: 'POST', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ ids: unreadIds }) }); if (!response.ok) throw new Error(`Erreur HTTP ${response.status}`); // Update local state this.insights = this.insights.map(i => ({ ...i, lu: true })); log.debug('STATE_UPDATE', { markedAsRead: unreadIds.length }); log.info('WORKFLOW_SUCCESS', { workflowId, count: unreadIds.length }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
