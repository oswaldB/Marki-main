dismissInsight: async function(insight) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'dismissInsight', insightId: insight?.id }); if (!confirm('Ignorer cette recommandation ?')) { log.info('WORKFLOW_CANCELLED', { workflowId, reason: 'user_cancelled' }); return; } this.loading = true; try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { endpoint: `/api/insights/${insight.id}/dismiss` }); const response = await fetch(`/api/insights/${insight.id}/dismiss`, { method: 'POST', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error(`Erreur HTTP ${response.status}`); // Remove from list this.insights = this.insights.filter(i => i.id !== insight.id); log.debug('STATE_UPDATE', { remainingCount: this.insights.length }); log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
