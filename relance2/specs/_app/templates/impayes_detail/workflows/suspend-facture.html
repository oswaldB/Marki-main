suspendFacture: async function(id) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'suspendFacture', impayeId: id }); if (!confirm('Suspendre cette facture ?')) return; this.loading = true; try { const token = localStorage.getItem('token'); const response = await fetch(`/api/impayes/${id}/suspendre`, { method: 'POST', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur lors de la suspension'); this.impaye.statut = 'suspendu'; log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
