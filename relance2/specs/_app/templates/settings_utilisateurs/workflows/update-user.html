updateUser: async function(id) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'updateUser', userId: id }); this.loading = true; try { const token = localStorage.getItem('token'); const user = this.users.find(u => u.id === id); const response = await fetch(`/api/users/${id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify(user) }); if (!response.ok) throw new Error('Erreur lors de la mise à jour'); user._editing = false; log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
