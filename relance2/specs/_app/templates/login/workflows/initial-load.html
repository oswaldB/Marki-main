initialLoad: async function() { const workflowId = crypto.randomUUID(); const startTime = Date.now(); log.info('WORKFLOW_START', { workflowId, workflow: 'initialLoad', page: 'login' }); try { // Étape 1: Vérification token existant log.debug('TOKEN_CHECK_START'); const token = localStorage.getItem('token'); if (token) { log.debug('TOKEN_FOUND', { tokenPreview: token.substring(0, 10) + '...' }); // Optionnel: vérifier si token encore valide log.debug('TOKEN_VALIDATION_START'); // Ici on pourrait faire un appel API pour vérifier log.debug('TOKEN_VALIDATION_COMPLETE'); // Redirection log.info('REDIRECT_DASHBOARD', { reason: 'existing_token' }); window.location.href = '/dashboard'; } else { log.debug('TOKEN_NOT_FOUND'); log.info('SHOW_LOGIN_FORM'); } log.info('WORKFLOW_SUCCESS', { workflowId, duration: Date.now() - startTime, hasToken: !!token }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message, stack: error.stack }); // En cas d'erreur, on affiche quand même le formulaire this.error = 'Erreur lors du chargement. Veuillez réessayer.'; } }
