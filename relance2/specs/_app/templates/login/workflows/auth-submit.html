authSubmit: async function() { const workflowId = crypto.randomUUID(); const startTime = Date.now(); log.info('WORKFLOW_START', { workflowId, workflow: 'authSubmit', username: this.form.username }); this.loading = true; this.clearErrors(); try { // Étape 1: Validation log.debug('VALIDATION_START'); if (!this.validate()) { log.warn('VALIDATION_FAILED', { errors: this.errors }); return; } log.debug('VALIDATION_SUCCESS'); // Étape 2: Appel API log.debug('API_CALL_START', { endpoint: '/api/auth/login', method: 'POST', username: this.form.username }); const response = await fetch('/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: this.form.username, password: this.form.password }) }); log.debug('API_CALL_COMPLETE', { status: response.status, ok: response.ok }); // Étape 3: Traitement réponse log.debug('RESPONSE_PROCESSING_START'); const data = await response.json(); if (!response.ok) { throw new Error(data.error || 'Identifiants invalides'); } log.debug('RESPONSE_PROCESSING_COMPLETE', { hasToken: !!data.token, user: data.user?.username || data.user?.email }); // Étape 4: Connexion réussie this.loginSuccess(data.token, data.user); log.info('WORKFLOW_SUCCESS', { workflowId, duration: Date.now() - startTime, user: data.user?.username || data.user?.email }); } catch (error) { this.error = error.message || 'Erreur de connexion'; log.error('WORKFLOW_ERROR', { workflowId, error: error.message, stack: error.stack, duration: Date.now() - startTime }); } finally { this.loading = false; } }
