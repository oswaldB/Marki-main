reglerFacture: async function() { const workflowId = crypto.randomUUID(); const startTime = Date.now(); log.info('WORKFLOW_START', { workflowId, workflow: 'reglerFacture', montant: this.mission?.montant }); this.loading = true; try { log.debug('API_CALL_START', { endpoint: '/api/portail/payer' }); const response = await fetch('/api/portail/payer', { method: 'POST', headers: { 'Authorization': `Bearer ${this.token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ mission_id: this.mission.id }) }); log.debug('API_RESPONSE_RECEIVED', { status: response.status }); if (!response.ok) { throw new Error('Erreur lors du paiement'); } const data = await response.json(); // Redirection vers la page de paiement if (data.payment_url) { window.location.href = data.payment_url; } log.info('WORKFLOW_SUCCESS', { workflowId, duration: Date.now() - startTime }); } catch (error) { this.error = error.message; log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } finally { this.loading = false; } }
