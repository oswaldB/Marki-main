copyLien: async function(lien) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'copyLien', lien }); try { await navigator.clipboard.writeText(lien); log.debug('CLIPBOARD_WRITE', { lien }); log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
