    </style>
</head>
<body>
    <div class="container" x-data="login()">
        <h1>login</h1>
        <p>Cell squelette fonctionnel</p>
        <div x-show="loading">Chargement...</div>
        <div x-show="error" style="color: red" x-text="error"></div>
        <button @click="loadData()">Tester</button>
        <pre x-show="data"><code x-text="JSON.stringify(data, null, 2)"></code></pre>
    </div>
    <script>
        function login() {
            return {
                loading: false,
                error: null,
                data: null,
                async loadData() {
                    this.loading = true;
                    this.error = null;
                    try {
                        const response = await fetch('/login/');
                        this.data = await response.json();
                    } catch (e) {
                        this.error = e.message;
                    } finally {
                        this.loading = false;
                    }
                }
            }
        }
    </script>
</body>
</html>
