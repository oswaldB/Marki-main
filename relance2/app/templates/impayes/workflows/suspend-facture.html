// Workflow: Suspendre une facture suspendFacture: async function(impaye) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'suspendFacture', impayeId: impaye.id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/impayes/${impaye.id}`, method: 'PUT' }); const response = await fetch(`/api/impayes/${impaye.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ suspendu: 1, suspend_motif: 'Suspension manuelle' }) }); if (!response.ok) throw new Error('Erreur lors de la suspension'); // Update local state impaye.suspendu = 1; impaye.suspend_motif = 'Suspension manuelle'; impaye.suspend_date = new Date().toISOString(); log.info('WORKFLOW_SUCCESS', { workflowId, impayeId: impaye.id }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, impayeId: impaye.id, error: error.message }); } }
