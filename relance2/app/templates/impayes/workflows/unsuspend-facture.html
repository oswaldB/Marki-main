// Workflow: Réactiver une facture suspendue unsuspendFacture: async function(impaye) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'unsuspendFacture', impayeId: impaye.id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/impayes/${impaye.id}`, method: 'PUT' }); const response = await fetch(`/api/impayes/${impaye.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ suspendu: 0 }) }); if (!response.ok) throw new Error('Erreur lors de la réactivation'); // Update local state impaye.suspendu = 0; impaye.suspend_motif = null; impaye.suspend_date = null; log.info('WORKFLOW_SUCCESS', { workflowId, impayeId: impaye.id }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, impayeId: impaye.id, error: error.message }); } }
