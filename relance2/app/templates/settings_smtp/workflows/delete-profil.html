// Workflow: Supprimer profil deleteProfil: async function(id) { const workflowId = crypto.randomUUID(); if (!confirm('Êtes-vous sûr de vouloir supprimer ce profil ?')) { log.info('WORKFLOW_CANCELLED', { workflowId, reason: 'user_cancelled' }); return; } log.info('WORKFLOW_START', { workflowId, workflow: 'deleteProfil', profilId: id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/settings/smtp/${id}` }); const response = await fetch(`/api/settings/smtp/${id}`, { method: 'DELETE', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur suppression'); log.info('WORKFLOW_SUCCESS', { workflowId }); await this.initialLoad(); if (typeof Alpine !== 'undefined' && Alpine.store('ui')) { Alpine.store('ui').addToast('Profil supprimé', 'success'); } } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
