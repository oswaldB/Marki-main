// Workflow: Changement de séquence de relance avec confirmation // Affiche une popup pour choisir la séquence puis entre 'restart' et 'continue' /** * Ferme le sélecteur de séquence et reset les valeurs */ closeSequenceSelector: function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'closeSequenceSelector' }); this.showSequenceSelector = false; this.pendingSequenceId = null; this.pendingSequenceName = null; this.error = null; log.info('WORKFLOW_SUCCESS', { workflowId }); }, /** * Confirme le changement de séquence avec le mode choisi * @param {string} mode - 'restart' ou 'continue' */ confirmChangeSequence: async function(mode) { const workflowId = crypto.randomUUID(); const startTime = Date.now(); log.info('WORKFLOW_START', { workflowId, workflow: 'confirmChangeSequence', mode, impayeId: this.impaye?.id, sequenceId: this.pendingSequenceId }); if (!this.impaye || !this.pendingSequenceId || this.pendingSequenceId === '__NONE__') { log.error('WORKFLOW_ERROR', { workflowId, error: 'No impaye or invalid pending sequence' }); return; } // Vérifier que ce n'est pas la même séquence if (this.pendingSequenceId === this.impaye.sequence_id) { this.error = 'Veuillez sélectionner une séquence différente de l\'actuelle'; return; } this.changingSequence = true; try { const token = localStorage.getItem('marki_token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/impayes/${this.impaye.id}/change-sequence`, method: 'POST', body: { sequence_id: this.pendingSequenceId, mode: mode } }); const response = await fetch(`/api/impayes/${this.impaye.id}/change-sequence`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify({ sequence_id: this.pendingSequenceId, mode: mode }) }); log.debug('API_RESPONSE_RECEIVED', { workflowId, status: response.status }); // Gestion token expiré if (response.status === 401) { log.warn('AUTH_TOKEN_EXPIRED', { workflowId }); localStorage.removeItem('marki_token'); localStorage.removeItem('marki_user'); window.location.href = '/login'; return; } const data = await response.json(); if (!data.success) { throw new Error(data.error?.message || 'Erreur lors du changement de séquence'); } // Mettre à jour l'impayé local this.impaye.sequence_id = this.pendingSequenceId; this.impaye.sequence_nom = data.sequence?.nom; this.selectedSequence = this.pendingSequenceId; log.debug('STATE_UPDATE', { workflowId, prop: 'impaye.sequence_id', value: this.pendingSequenceId }); // Recharger les relances depuis l'API if (this.impaye.payer_id) { await this.loadRelances(this.impaye.payer_id); } // Rafraîchir l'historique await this.loadEvents(this.impaye.id); this.mergeHistorique(); // Fermer la popup this.showSequenceSelector = false; this.pendingSequenceId = null; this.pendingSequenceName = null; // Toast de succès const modeLabel = mode === 'restart' ? 'Séquence recommencée à zéro' : 'Séquence continuée'; this.showToast( 'Succès', `${modeLabel}. ${data.relances_created || 0} relance(s) créée(s).`, 'success' ); log.info('WORKFLOW_SUCCESS', { workflowId, mode, relancesCreated: data.relances_created, relancesDeleted: data.relances_deleted, duration: Date.now() - startTime }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); this.error = error.message; } finally { this.changingSequence = false; } }, /** * Retire l'impayé de toutes les séquences (sequence_id = null) */ confirmRemoveFromSequences: async function() { const workflowId = crypto.randomUUID(); const startTime = Date.now(); log.info('WORKFLOW_START', { workflowId, workflow: 'confirmRemoveFromSequences', impayeId: this.impaye?.id }); if (!this.impaye) { log.error('WORKFLOW_ERROR', { workflowId, error: 'No impaye loaded' }); return; } if (!this.impaye.sequence_id) { this.error = 'Cet impayé n\'est déjà dans aucune séquence'; return; } this.changingSequence = true; try { const token = localStorage.getItem('marki_token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/impayes/${this.impaye.id}/change-sequence`, method: 'POST', body: { remove_all: true } }); const response = await fetch(`/api/impayes/${this.impaye.id}/change-sequence`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` }, body: JSON.stringify({ remove_all: true }) }); log.debug('API_RESPONSE_RECEIVED', { workflowId, status: response.status }); // Gestion token expiré if (response.status === 401) { log.warn('AUTH_TOKEN_EXPIRED', { workflowId }); localStorage.removeItem('marki_token'); localStorage.removeItem('marki_user'); window.location.href = '/login'; return; } const data = await response.json(); if (!data.success) { throw new Error(data.error?.message || 'Erreur lors du retrait des séquences'); } // Mettre à jour l'impayé local this.impaye.sequence_id = null; this.impaye.sequence_nom = null; this.selectedSequence = ''; log.debug('STATE_UPDATE', { workflowId, prop: 'impaye.sequence_id', value: null }); // Recharger les relances depuis l'API if (this.impaye.payer_id) { await this.loadRelances(this.impaye.payer_id); } // Rafraîchir l'historique await this.loadEvents(this.impaye.id); this.mergeHistorique(); // Fermer la popup this.showSequenceSelector = false; this.pendingSequenceId = null; this.pendingSequenceName = null; // Toast de succès this.showToast( 'Succès', `Retiré de toutes les séquences. ${data.relances_deleted || 0} relance(s) supprimée(s).`, 'success' ); log.info('WORKFLOW_SUCCESS', { workflowId, relancesDeleted: data.relances_deleted, duration: Date.now() - startTime }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); this.error = error.message; } finally { this.changingSequence = false; } }
