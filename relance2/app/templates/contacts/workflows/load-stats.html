// ===================================================== // WORKFLOW: load-stats - Chargement des statistiques // ===================================================== loadStatsOnly: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'loadStatsOnly', page: 'contacts' }); try { log.debug('API_CALL_START', { endpoint: '/api/contacts/stats' }); const response = await fetch('/api/contacts/stats'); log.debug('API_RESPONSE_RECEIVED', { status: response.status }); if (!response.ok) { throw new Error(`Erreur HTTP ${response.status}`); } const statsData = await response.json(); log.debug('STATE_UPDATE', { oldStats: this.stats, newStats: statsData }); this.stats = statsData; log.info('WORKFLOW_SUCCESS', { workflowId, workflow: 'loadStatsOnly', stats: statsData }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, workflow: 'loadStatsOnly', error: error.message }); this.stats = { total: 0, entreprises: 0, personnes: 0, avecImpayes: 0, blacklist: 0, sansEmail: 0 }; } }
