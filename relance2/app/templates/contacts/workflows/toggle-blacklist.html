// Workflow: Toggle blacklist contact toggleBlacklist: async function(contact) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'toggleBlacklist', contactId: contact.id }); // Optimistic update const previousStatus = contact.is_blacklisted; contact.is_blacklisted = !previousStatus; try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/contacts/${contact.id}`, method: 'PUT' }); const response = await fetch(`/api/contacts/${contact.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ is_blacklisted: contact.is_blacklisted, blacklist_motif: contact.is_blacklisted ? 'Blacklist manuelle' : null }) }); if (!response.ok) throw new Error('Erreur serveur'); const data = await response.json(); // Update avec données reçues contact.blacklist_date = data.blacklist_date; contact.blacklist_motif = data.blacklist_motif; log.info('WORKFLOW_SUCCESS', { workflowId, contactId: contact.id, is_blacklisted: data.is_blacklisted }); // Notification this.showNotification && this.showNotification( `Contact ${contact.is_blacklisted ? 'blacklisté' : 'déblacklisté'}`, 'success' ); } catch (error) { // Rollback contact.is_blacklisted = previousStatus; log.error('WORKFLOW_ERROR', { workflowId, contactId: contact.id, error: error.message }); this.showNotification && this.showNotification( 'Erreur lors de la mise à jour', 'error' ); } }
