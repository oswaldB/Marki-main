// Workflow: Export données exportData: function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'exportData' }); // CSV export const headers = ['Nom', 'Prénom', 'Email', 'Téléphone', 'Type']; const rows = this.filteredContacts.map(c => [ c.nom, c.prenom, c.email, c.telephone, c.type_personne ]); const csv = [headers, ...rows].map(row => row.join(';')).join('\n'); const blob = new Blob([csv], { type: 'text/csv' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = `contacts_${new Date().toISOString().split('T')[0]}.csv`; a.click(); URL.revokeObjectURL(url); log.info('WORKFLOW_SUCCESS', { workflowId, exported: rows.length }); }
