// Workflow: Supprimer relance supprimerRelance: async function(id) { const workflowId = crypto.randomUUID(); if (!confirm('Êtes-vous sûr de vouloir supprimer cette relance ?')) { log.info('WORKFLOW_CANCELLED', { workflowId, reason: 'user_cancelled' }); return; } log.info('WORKFLOW_START', { workflowId, workflow: 'supprimerRelance', relanceId: id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/relances/${id}` }); const response = await fetch(`/api/relances/${id}`, { method: 'DELETE', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur suppression'); log.info('WORKFLOW_SUCCESS', { workflowId }); // Recharger les données await this.initialLoad(); // Notification if (typeof Alpine !== 'undefined' && Alpine.store('ui')) { Alpine.store('ui').addToast('Relance supprimée', 'success'); } } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); if (typeof Alpine !== 'undefined' && Alpine.store('ui')) { Alpine.store('ui').addToast(error.message, 'error'); } } }
