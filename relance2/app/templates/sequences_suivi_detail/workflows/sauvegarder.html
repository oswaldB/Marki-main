// Workflow: Save suivi sequence changes sauvegarder: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'sauvegarder' }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/sequences/${this.sequence.id}`, method: 'PUT' }); const response = await fetch(`/api/sequences/${this.sequence.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify(this.sequence) }); if (!response.ok) throw new Error('Erreur lors de la sauvegarde'); log.info('WORKFLOW_SUCCESS', { workflowId }); this.showNotification && this.showNotification('Séquence sauvegardée', 'success'); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
