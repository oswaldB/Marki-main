// Workflow: Delete email from suivi sequence supprimerEmail: async function(email) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'supprimerEmail', emailId: email.id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/sequences/emails/${email.id}`, method: 'DELETE' }); const response = await fetch(`/api/sequences/emails/${email.id}`, { method: 'DELETE', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur lors de la suppression'); this.emails = this.emails.filter(e => e.id !== email.id); log.info('WORKFLOW_SUCCESS', { workflowId, emailId: email.id }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, emailId: email.id, error: error.message }); } }
