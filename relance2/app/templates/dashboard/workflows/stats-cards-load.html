/** * Workflow: Stats Cards Load * Calcule et affiche les KPIs des cartes statistiques * * @action Calculer les KPIs principaux * @checkpoint kpis-calculated */ async loadStatsCards() { // @checkpoint kpis-start this.calculateKPIs(); // @checkpoint kpis-end }, calculateKPIs() { const nowStats = new Date(); const twelveMonthsAgo = new Date(); twelveMonthsAgo.setMonth(twelveMonthsAgo.getMonth() - 12); // Helpers const isEchu = (dateStr) => { if (!dateStr) return false; return new Date(dateStr) < nowStats; }; const isWithinLast12Months = (dateStr) => { if (!dateStr) return false; const date = new Date(dateStr); return date >= twelveMonthsAgo && date <= nowStats; }; const getDays = (dateStr) => { if (!dateStr) return 0; return Math.floor((nowStats - new Date(dateStr)) / (1000 * 60 * 60 * 24)); }; // KPIs principaux this.kpis.facturesEnAttente = this.impayes.filter(i => i.reste_a_payer > 0).length; // Factures échues = statut impaye + reste_a_payer > 0 + date échue this.kpis.impayesActifs = this.impayes.filter(i => i.statut === 'impaye' && i.reste_a_payer > 0 && isEchu(i.date_echeance) ).length; // Montant total uniquement sur factures échues const facturesEchues = this.impayes.filter(i => i.reste_a_payer > 0 && isEchu(i.date_echeance)); this.kpis.montantTotal = facturesEchues.reduce((sum, i) => sum + (i.reste_a_payer || 0), 0); // DSO (Days Sales Outstanding) = moyenne des jours de retard pondérée par le montant // Formule: Σ(jours de retard × montant) / Σ(montant) let totalPondere = 0; facturesEchues.forEach(i => { const days = getDays(i.date_echeance); const montant = i.reste_a_payer || 0; totalPondere += days * montant; }); this.kpis.dso = this.kpis.montantTotal > 0 ? Math.round(totalPondere / this.kpis.montantTotal) : 0; this.kpis.relancesJour = this.relances.length; // Taux de recouvrement du mois dernier et évolution vs mois précédent // Calculer le taux pour un mois donné (uniquement sur factures échues) const getTauxForMonth = (year, month) => { const startOfMonth = new Date(year, month, 1); const endOfMonth = new Date(year, month + 1, 0); const facturesDuMois = this.allFactures12M.filter(i => { if (!i.date_echeance) return false; const date = new Date(i.date_echeance); // Filtrer sur le mois ET sur les factures échues return date >= startOfMonth && date <= endOfMonth && isEchu(i.date_echeance); }); if (facturesDuMois.length === 0) return null; const soldes = facturesDuMois.filter(i => (i.reste_a_payer || 0) === 0).length; return Math.round((soldes / facturesDuMois.length) * 100); }; // Mois dernier (M-1) const lastMonth = new Date(nowStats.getFullYear(), nowStats.getMonth() - 1, 1); const tauxLastMonth = getTauxForMonth(lastMonth.getFullYear(), lastMonth.getMonth()); // Mois d'avant (M-2) const prevMonth = new Date(nowStats.getFullYear(), nowStats.getMonth() - 2, 1); const tauxPrevMonth = getTauxForMonth(prevMonth.getFullYear(), prevMonth.getMonth()); if (tauxLastMonth !== null) { this.kpis.tauxRecouvrement = tauxLastMonth; if (tauxPrevMonth !== null) { this.kpis.tauxRecouvrementDiff = tauxLastMonth - tauxPrevMonth; } else { this.kpis.tauxRecouvrementDiff = null; } } else { // Pas assez de données pour le mois dernier, on n'affiche pas this.kpis.tauxRecouvrement = null; this.kpis.tauxRecouvrementDiff = null; } // Reset anciennete this.kpis.anciennete = { moins7j: 0, moins7jMontant: 0, j8a30: 0, j8a30Montant: 0, j31a60: 0, j31a60Montant: 0, j60a120: 0, j60a120Montant: 0, plus120j: 0, plus120jMontant: 0 }; // Calcul ancienneté uniquement sur factures échues this.impayes.filter(i => isEchu(i.date_echeance)).forEach(i => { const days = getDays(i.date_echeance); const montant = i.reste_a_payer || 0; if (days < 7) { this.kpis.anciennete.moins7j++; this.kpis.anciennete.moins7jMontant += montant; } else if (days <= 30) { this.kpis.anciennete.j8a30++; this.kpis.anciennete.j8a30Montant += montant; } else if (days <= 60) { this.kpis.anciennete.j31a60++; this.kpis.anciennete.j31a60Montant += montant; } else if (days <= 120) { this.kpis.anciennete.j60a120++; this.kpis.anciennete.j60a120Montant += montant; } else { this.kpis.anciennete.plus120j++; this.kpis.anciennete.plus120jMontant += montant; } }); }
