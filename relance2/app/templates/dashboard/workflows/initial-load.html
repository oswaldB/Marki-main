} // Charger le dernier event de synchro log.debug('API_CALL_START', { workflowId, endpoint: '/api/events?type=sync&limit=1' }); const syncResponse = await fetch('/api/events?type=sync&limit=1', { headers: { 'Authorization': `Bearer ${token}` } }); this.calculateKPIs(); log.debug('CALCULATE_KPIS_COMPLETE', { workflowId }); const duration = Date.now() - startTime; log.info('WORKFLOW_SUCCESS', { workflowId, duration }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); this.error = error.message; } finally { this.loading = false; } }, // Calcul des KPIs côté frontend (selon specs) calculateKPIs: function() { const now = new Date(); // KPI 1: Factures en attente (reste_a_payer > 0) const facturesEnAttente = this.impayes.filter(i => i.reste_a_payer > 0);
