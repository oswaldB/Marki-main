// Workflow: Apply AI insight applyInsight: async function(insight) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'applyInsight', insightId: insight.id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/smart-marki/insights/${insight.id}/apply`, method: 'POST' }); const response = await fetch(`/api/smart-marki/insights/${insight.id}/apply`, { method: 'POST', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur lors de l\'application'); // Mark as applied insight.applied = true; log.info('WORKFLOW_SUCCESS', { workflowId, insightId: insight.id }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, insightId: insight.id, error: error.message }); } }
