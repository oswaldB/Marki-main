// Workflow: Change sequence for impaye changerSequence: async function(sequenceId) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'changerSequence', sequenceId }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/impayes/${this.impaye.id}`, method: 'PUT' }); const response = await fetch(`/api/impayes/${this.impaye.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ sequence_id: sequenceId }) }); if (!response.ok) throw new Error('Erreur lors du changement de séquence'); // Update local state this.impaye.sequence_id = sequenceId; log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
