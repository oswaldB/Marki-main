// Workflow: Toggle email active status toggleEmail: async function(email) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'toggleEmail', emailId: email.id }); try { const token = localStorage.getItem('token'); const newValue = !email.actif; log.debug('API_CALL_START', { workflowId, endpoint: `/api/sequences/emails/${email.id}`, method: 'PUT' }); const response = await fetch(`/api/sequences/emails/${email.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ actif: newValue }) }); if (!response.ok) throw new Error('Erreur lors du changement'); email.actif = newValue; log.info('WORKFLOW_SUCCESS', { workflowId, emailId: email.id, newValue }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, emailId: email.id, error: error.message }); } }
