// Workflow: Initial load for sequence detail initialLoad: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'initialLoad' }); try { const token = localStorage.getItem('token'); const sequenceId = this.getSequenceIdFromUrl(); log.debug('API_CALL_START', { workflowId, endpoint: `/api/sequences/${sequenceId}`, method: 'GET' }); const response = await fetch(`/api/sequences/${sequenceId}`, { headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur chargement séquence'); const data = await response.json(); this.sequence = data.sequence; this.emails = data.emails || []; log.info('WORKFLOW_SUCCESS', { workflowId, sequenceId }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); this.error = error.message; } }
