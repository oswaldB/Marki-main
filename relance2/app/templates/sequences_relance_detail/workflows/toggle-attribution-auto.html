// Workflow: Toggle automatic attribution toggleAttributionAuto: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'toggleAttributionAuto' }); try { const token = localStorage.getItem('token'); const newValue = !this.sequence?.attribution_auto; log.debug('API_CALL_START', { workflowId, endpoint: `/api/sequences/${this.sequence.id}`, method: 'PUT' }); const response = await fetch(`/api/sequences/${this.sequence.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ attribution_auto: newValue }) }); if (!response.ok) throw new Error('Erreur lors du changement'); this.sequence.attribution_auto = newValue; log.info('WORKFLOW_SUCCESS', { workflowId, newValue }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
