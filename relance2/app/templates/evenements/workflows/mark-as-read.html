// Workflow: Marquer comme lu markAsRead: async function(id) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'markAsRead', eventId: id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/events/${id}/read` }); const response = await fetch(`/api/events/${id}/read`, { method: 'POST', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur'); // Mettre à jour localement const event = this.events.find(e => e.id === id); if (event) event.read = 1; log.info('WORKFLOW_SUCCESS', { workflowId }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
