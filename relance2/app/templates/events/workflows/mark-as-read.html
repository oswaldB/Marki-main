// Workflow: Marquer un événement comme lu /** * @action Marquer un événement spécifique comme lu * @param {Object} event - L'événement à marquer comme lu * @checkpoint event-marked-read */ markAsRead: function(event) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'markAsRead', eventId: event?.id }); if (!event || !event.id) { log.warn('INVALID_EVENT', { workflowId }); return; } // @action Trouver l'index de l'event const eventIndex = this.events.findIndex(e => e.id === event.id); if (eventIndex === -1) { log.warn('EVENT_NOT_FOUND', { workflowId, eventId: event.id }); return; } // @action Vérifier si l'event était déjà lu const wasUnread = !this.events[eventIndex].read; if (!wasUnread) { log.debug('ALREADY_READ', { workflowId, eventId: event.id }); return; } // @action Mettre à jour l'état (optimistic update) this.events[eventIndex].read = true; log.debug('EVENT_MARKED_READ', { workflowId, eventId: event.id, wasUnread: wasUnread }); // @action Sauvegarder dans localStorage this.saveEventsToStorage(); log.debug('EVENTS_SAVED_TO_STORAGE', { workflowId }); log.info('WORKFLOW_SUCCESS', { workflowId, eventId: event.id, remainingUnread: this.unreadCount }); }, /** * @action Basculer le statut lu/non lu d'un événement * @param {Object} event - L'événement à basculer * @checkpoint event-toggled */ toggleRead: function(event) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'toggleRead', eventId: event?.id, currentReadStatus: event?.read }); if (!event || !event.id) { log.warn('INVALID_EVENT', { workflowId }); return; } // @action Trouver l'index de l'event const eventIndex = this.events.findIndex(e => e.id === event.id); if (eventIndex === -1) { log.warn('EVENT_NOT_FOUND', { workflowId, eventId: event.id }); return; } // @action Basculer le statut const newStatus = !this.events[eventIndex].read; this.events[eventIndex].read = newStatus; log.debug('EVENT_TOGGLED', { workflowId, eventId: event.id, newStatus: newStatus }); // @action Sauvegarder dans localStorage this.saveEventsToStorage(); log.info('WORKFLOW_SUCCESS', { workflowId, eventId: event.id, newStatus: newStatus, unreadCount: this.unreadCount }); }
