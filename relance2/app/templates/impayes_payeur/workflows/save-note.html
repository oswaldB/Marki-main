// Workflow: Save note saveNote: async function(impaye) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'saveNote', impayeId: impaye.id }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: `/api/impayes/${impaye.id}`, method: 'PUT' }); const response = await fetch(`/api/impayes/${impaye.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ notes: impaye.notes }) }); if (!response.ok) throw new Error('Erreur lors de la sauvegarde'); log.info('WORKFLOW_SUCCESS', { workflowId, impayeId: impaye.id }); this.showNotification && this.showNotification('Note sauvegardée', 'success'); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, impayeId: impaye.id, error: error.message }); } }
