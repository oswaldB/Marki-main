// Workflow: Sync data syncData: async function() { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'syncData' }); try { const token = localStorage.getItem('token'); log.debug('API_CALL_START', { workflowId, endpoint: '/api/impayes/sync', method: 'POST' }); const response = await fetch('/api/impayes/sync', { method: 'POST', headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) throw new Error('Erreur lors de la synchronisation'); await this.initialLoad(); log.info('WORKFLOW_SUCCESS', { workflowId }); this.showNotification && this.showNotification('Données synchronisées', 'success'); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
