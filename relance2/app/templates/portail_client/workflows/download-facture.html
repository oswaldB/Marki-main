// Workflow: Download facture PDF downloadFacture: function(facture) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'downloadFacture', factureId: facture.id }); try { if (facture.url_pdf_token) { window.open(`/api/portail/factures/${facture.id}/pdf?token=${facture.url_pdf_token}`, '_blank'); } else { log.warn('PDF_NOT_AVAILABLE', { workflowId, factureId: facture.id }); } log.info('WORKFLOW_SUCCESS', { workflowId, factureId: facture.id }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
