// Workflow: Soumission authentification authSubmit: async function() { const workflowId = crypto.randomUUID(); const startTime = Date.now(); log.info('WORKFLOW_START', { workflowId, workflow: 'authSubmit', page: 'login' }); // 1. Réinitialiser les erreurs this.errors = { username: null, password: null }; this.error = null; // 2. Validation log.debug('VALIDATION_START', { workflowId, form: { username: this.form.username } }); let hasError = false; if (!this.form.username) { this.errors.username = 'Ce champ est requis'; hasError = true; log.debug('VALIDATION_ERROR', { workflowId, field: 'username' }); } if (!this.form.password) { this.errors.password = 'Ce champ est requis'; hasError = true; log.debug('VALIDATION_ERROR', { workflowId, field: 'password' }); } if (hasError) { log.info('VALIDATION_FAILED', { workflowId }); return; } log.info('VALIDATION_COMPLETE', { workflowId }); // 3. Appel API this.loading = true; try { log.debug('API_CALL_START', { workflowId, endpoint: '/api/auth/login', method: 'POST' }); const response = await fetch('/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: this.form.username, password: this.form.password }) }); log.debug('API_RESPONSE_RECEIVED', { workflowId, status: response.status }); if (!response.ok) { const data = await response.json(); throw new Error(data.error || 'Identifiants invalides'); } const data = await response.json(); log.info('API_CALL_COMPLETE', { workflowId, userId: data.user?.id }); // 4. Stockage du token localStorage.setItem('token', data.token); log.debug('TOKEN_SAVED', { workflowId }); // 5. Redirection const duration = Date.now() - startTime; log.info('WORKFLOW_SUCCESS', { workflowId, duration, redirect: '/dashboard' }); window.location.href = '/dashboard'; } catch (error) { const duration = Date.now() - startTime; this.error = error.message || 'Échec de la connexion'; log.error('WORKFLOW_ERROR', { workflowId, error: error.message, duration }); } finally { this.loading = false; } }
