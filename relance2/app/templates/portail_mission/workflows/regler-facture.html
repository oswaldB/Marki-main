// Workflow: Pay facture from mission reglerFacture: async function(facture) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'reglerFacture', factureId: facture.id }); try { const token = localStorage.getItem('portail_token'); log.debug('API_CALL_START', { workflowId, endpoint: '/api/portail/mission/pay', method: 'POST' }); const response = await fetch('/api/portail/mission/pay', { method: 'POST', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ facture_id: facture.id }) }); if (!response.ok) throw new Error('Erreur lors du paiement'); facture.statut = 'paye'; log.info('WORKFLOW_SUCCESS', { workflowId, factureId: facture.id }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, factureId: facture.id, error: error.message }); } }
