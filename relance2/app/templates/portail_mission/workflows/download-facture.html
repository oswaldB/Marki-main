// Workflow: Download facture from mission downloadFacture: function(facture) { const workflowId = crypto.randomUUID(); log.info('WORKFLOW_START', { workflowId, workflow: 'downloadFacture', factureId: facture.id }); try { if (facture.pdf_url) { window.open(facture.pdf_url, '_blank'); } log.info('WORKFLOW_SUCCESS', { workflowId, factureId: facture.id }); } catch (error) { log.error('WORKFLOW_ERROR', { workflowId, error: error.message }); } }
