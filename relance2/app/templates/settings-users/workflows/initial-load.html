/** * @action Initialiser les états * @checkpoint state-initialized */ async init() { await this.loadUsers(); }, /** * @action Charger la liste des utilisateurs * @checkpoint users-fetched * @checkpoint data-stored * @checkpoint loading-complete */ async loadUsers() { this.loading = true; this.error = null; try { const token = localStorage.getItem('marki_token'); if (!token) { window.location.href = '/login'; return; } // Détecter si admin depuis le token const payload = JSON.parse(atob(token.split('.')[1])); this.isAdmin = payload.role === 'admin'; const response = await fetch('/api/users', { headers: { 'Authorization': `Bearer ${token}` } }); if (!response.ok) { throw new Error('Erreur lors du chargement des utilisateurs'); } const data = await response.json(); if (!data.success) { throw new Error(data.error?.message || 'Erreur de chargement'); } this.users = data.users || []; } catch (err) { this.error = err.message; console.error('[USERS] Error:', err); } finally { this.loading = false; } }
