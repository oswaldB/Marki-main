async updateSequence(event) { // @checkpoint update-sequence-called const newSequence = event.target.value; try { const token = localStorage.getItem('marki_token'); if (!token) { window.location.href = '/login'; return; } // @action Appeler API PUT // @checkpoint api-called const response = await fetch(`/api/impayes/${this.selectedFacture.id}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ sequence_code: newSequence, updated_at: new Date().toISOString() }) }); if (!response.ok) { throw new Error(`Erreur ${response.status}`); } const data = await response.json(); if (!data.success) { throw new Error(data.error?.message || 'Erreur de mise à jour'); } // @action Mettre à jour local // @checkpoint local-updated const index = this.allFactures.findIndex(f => f.id === this.selectedFacture.id); if (index !== -1) { this.allFactures[index].sequence_code = newSequence; this.selectedFacture.sequence_code = newSequence; } // @action Notifier succès if (Alpine.store('ui')?.addToast) { Alpine.store('ui').addToast('Séquence mise à jour', 'success'); } log.info('sequence-updated', { factureId: this.selectedFacture.id, sequence: newSequence }); } catch (err) { log.error('update-sequence-error', err.message); if (Alpine.store('ui')?.addToast) { Alpine.store('ui').addToast(err.message, 'error'); } } }
