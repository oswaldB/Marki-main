/** * Workflow: Top Débiteurs Load * Calcule et affiche le top 5 des débiteurs * * @action Calculer top débiteurs * @checkpoint top-debiteurs-calculated */ async loadTopDebtors() { this.calculateTopDebtors(); // @checkpoint top-debiteurs-ready }, calculateTopDebtors() { const debtorsMap = {}; this.impayes.forEach(i => { const id = i.payer_id; if (!id) return; if (!debtorsMap[id]) { // Construction du nom depuis la jointure contacts (pas de champ dénormalisé) const prenom = i.payeur_contact_prenom || ''; const nom = i.payeur_contact_nom || ''; const name = (prenom + ' ' + nom).trim() || 'Inconnu'; debtorsMap[id] = { id, name, initials: this.getInitials(name), montant: 0, jours: 0, impayesCount: 0 }; } debtorsMap[id].montant += (i.reste_a_payer || 0); debtorsMap[id].impayesCount++; // Calcul des jours de retard basé sur la date d'échéance const days = Math.floor((new Date() - new Date(i.date_echeance)) / (1000 * 60 * 60 * 24)); if (days > debtorsMap[id].jours) debtorsMap[id].jours = days; }); this.topDebtors = Object.values(debtorsMap) .sort((a, b) => b.montant - a.montant) .slice(0, 5); }
