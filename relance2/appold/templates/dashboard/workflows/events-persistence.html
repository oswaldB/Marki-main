/** * Workflow: Events Persistence * Gère le stockage local des événements (max 100) * avec système de "marquer comme lu" * * @action Charger depuis localStorage * @action Sauvegarder dans localStorage * @action Marquer comme lu */ // Charger les événements depuis localStorage loadEventsFromStorage() { const EVENTS_STORAGE_KEY = 'marki_events'; const MAX_EVENTS = 100; try { const stored = localStorage.getItem(EVENTS_STORAGE_KEY); if (stored) { const events = JSON.parse(stored); // Garder uniquement les MAX_EVENTS plus récents return events.slice(0, MAX_EVENTS); } } catch (e) { console.error('[EVENTS] Erreur chargement localStorage:', e); } return []; }, // Sauvegarder les événements dans localStorage saveEventsToStorage() { const EVENTS_STORAGE_KEY = 'marki_events'; const MAX_EVENTS = 100; try { // Trier par date décroissante (plus récents d'abord) const sorted = [...this.events].sort((a, b) => { return new Date(b.created_at || 0) - new Date(a.created_at || 0); }); // Garder les MAX_EVENTS plus récents const toStore = sorted.slice(0, MAX_EVENTS); localStorage.setItem(EVENTS_STORAGE_KEY, JSON.stringify(toStore)); } catch (e) { console.error('[EVENTS] Erreur sauvegarde localStorage:', e); } }, // Marquer tous les événements comme lus markAllEventsAsRead() { this.events = this.events.map(e => ({ ...e, read: true })); this.saveEventsToStorage(); }, // Vider complètement le stockage local (suppression définitive) removeAllEvents() { const EVENTS_STORAGE_KEY = 'marki_events'; this.events = []; try { localStorage.removeItem(EVENTS_STORAGE_KEY); } catch (e) { console.error('[EVENTS] Erreur suppression localStorage:', e); } }, // Compter les événements non lus get unreadEventsCount() { return this.events.filter(e => !e.read).length; }
