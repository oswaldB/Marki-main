/** * Workflow: Events Load * Charge et formate les événements récents * * @action Charger les événements * @checkpoint events-loaded */ async loadEvents() { // @checkpoint auth-check const token = localStorage.getItem('marki_token'); if (!token) return; // @action Charger depuis localStorage d'abord // @checkpoint events-from-storage const storedEvents = this.loadEventsFromStorage(); if (storedEvents.length > 0) { this.events = this.formatEvents(storedEvents); } const headers = { 'Authorization': `Bearer ${token}` }; try { // @action Récupérer les événements depuis l'API (page 1, 100 par page) // @checkpoint events-fetch const eventsRes = await fetch('/api/events?page=1&per_page=100', { headers }); if (eventsRes.ok) { const eventsData = await eventsRes.json(); const freshEvents = eventsData.events || eventsData; // Compatibilité ancienne API // Marquer les nouveaux events comme non lus par défaut const newEventsWithReadStatus = freshEvents.map(e => ({ ...e, read: false })); // Fusionner avec les events existants (éviter les doublons par ID) const existingIds = new Set(this.events.map(e => e.id)); const newEvents = newEventsWithReadStatus.filter(e => !existingIds.has(e.id)); // Ajouter les nouveaux events au début this.events = [...this.formatEvents(newEvents), ...this.events]; // @action Sauvegarder dans localStorage (max 100) // @checkpoint events-saved this.saveEventsToStorage(); } // @action Récupérer dernier event de sync ou import // @checkpoint last-sync-fetch const syncRes = await fetch('/api/events?type=sync&page=1&per_page=1', { headers }); const importRes = await fetch('/api/events?type=import&page=1&per_page=1', { headers }); let lastEvent = null; if (syncRes.ok) { const syncData = await syncRes.json(); const syncEvents = syncData.events || syncData; if (syncEvents && syncEvents.length > 0) { lastEvent = syncEvents[0]; } } if (importRes.ok) { const importData = await importRes.json(); const importEvents = importData.events || importData; if (importEvents && importEvents.length > 0) { if (!lastEvent || new Date(importEvents[0].created_at) > new Date(lastEvent.created_at)) { lastEvent = importEvents[0]; } } } if (lastEvent) { this.lastSyncTime = this.formatDateTime(lastEvent.created_at); } // @checkpoint events-complete } catch (err) { console.error('[EVENTS] Error:', err); } }
